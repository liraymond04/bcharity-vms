Function useBookmark

  • useBookmark(params): UseBookmarkReturn

  • A hook to handle bookmark-related functionality

    Parameters

    • params: UseBookmarkParams

      the params for the hook

    Returns UseBookmarkReturn

    Example

    Bookmark example

    // Adapted from BookmarkButton.tsx
    const { bookmarked, error, isLoading, addBookmark, removeBookmark } =
      useBookmark({
        postTag,
        profile: currentUser,
        publicationId
      })

    return (
      <button
        disabled={isLoading}
        onClick={() => {
          if (bookmarked) {
            removeBookmark()
          } else {
            addBookmark()
          }
        }}
      >
        {isLoading ? (
          <Spinner size="sm" />
        ) : (
          <div className="transition duration-100 hover:scale-125">
            {bookmarked ? (
              <BookmarkSolid className="w-6 inline mb-1" />
            ) : (
              <BookmarkIcon className="w-6 inline mb-1" />
            )}
          </div>
        )}
      </button>
    )
